```java
public class Gugudan_3 {
	public static void main(String[] args) {
		for(int i=2;i<10;i+=3){
			for(int j=0;j<10;j++){
				for(int k=0;k<3;k++){ // 3 단씩 나누어 줌
					if(j==0) System.out.printf("%d단", i+k); // 몇 단인지 출력해주기 위한 조건
					else System.out.printf("%2d * %d = %2d ", i+k, j, (i+k)*j);
				}System.out.println();
			}
		}
	}
}
```
